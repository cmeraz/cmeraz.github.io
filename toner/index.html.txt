<script>
// Datos de prueba
const articulos = [
    { modelo: "HP 85A", impresoras: "HP LaserJet Pro P1102, P1102w, M1132, M1212nf" },
    { modelo: "Canon 725", impresoras: "Canon LBP6000, LBP6030, MF3010" },
    { modelo: "Brother TN-450", impresoras: "Brother HL-2240, HL-2270DW, MFC-7360N" },
    { modelo: "Samsung MLT-D111S", impresoras: "Samsung Xpress M2020, M2070" },
    { modelo: "Epson 664", impresoras: "Epson L100, L110, L120, L200, L210" }
];

// Elementos del DOM
const inputBusqueda = document.getElementById("busqueda");
const resultadoDiv = document.getElementById("resultado");

// Función para mostrar resultados
function mostrarResultados(lista) {
    resultadoDiv.innerHTML = "";
    if (lista.length === 0) {
        resultadoDiv.innerHTML = `<p class="no-resultados">No se encontraron resultados.</p>`;
        return;
    }
    lista.forEach(item => {
        const card = document.createElement("div");
        card.classList.add("card");
        card.innerHTML = `
            <div class="modelo">${item.modelo}</div>
            <div class="impresoras">${item.impresoras}</div>
        `;
        resultadoDiv.appendChild(card);
    });
}

// Función de búsqueda
function buscarArticulos(texto) {
    const busqueda = texto.toLowerCase();
    const filtrados = articulos.filter(a =>
        a.modelo.toLowerCase().includes(busqueda) ||
        a.impresoras.toLowerCase().includes(busqueda)
    );
    mostrarResultados(filtrados);
}

// Evento de búsqueda en tiempo real
inputBusqueda.addEventListener("input", e => {
    buscarArticulos(e.target.value);
});

// Mostrar todo al inicio
mostrarResultados(articulos);
</script>
